#To create dedicated service accounts, following least-privilege principles, that will gain access to certain processes and operations in google cloud 

#Three commands to create a cicd deployer to handle CI/CD deploymets, have access to manage VMs, access to push docker images to artifact registry, and for google's cloud build to be able to deploy docker containers when needed 
1. gcloud iam service-accounts create cicd-deployer --display-name="CI/CD Deployer" — Creates a robot account whose only job is to handle CI/CD deployments. After this runs, GCP creates an identity: cicd-deployer@security-hardening-lab-498320.iam.gserviceaccount.com. At this point it exists but has zero permissions — it can't do anything yet.

2. gcloud projects add-iam-policy-binding security-hardening-lab-498320 \
--member="serviceAccount:cicd-deployer@security-hardening-lab-498320.iam.gserviceaccount.com" \
--role="roles/compute.instanceAdmin.v1"

3. gcloud projects add-iam-policy-binding security-hardening-lab-498320 \
--member="serviceAccount:cicd-deployer@security-hardening-lab-498320.iam.gserviceaccount.com" \
--role="roles/artifactregistry.writer"

#Two commands to create VM runtime service account which will run on the VM, and grant it permission to read secrets from secret manager to grab TLS certificates, private keys, and other sensitive configs
1. gcloud iam service-accounts create vm-runtime \
--display-name="VM Runtime Account"

2. gcloud projects add-iam-policy-binding security-hardening-lab-498320 \
--member="serviceAccount:vm-runtime@security-hardening-lab-498320.iam.gserviceaccount.com" \
--role="roles/secretmanager.secretAccessor"
